Returns start and end times, plus duration in hours for all shifts
***************************************************************/
                                                             

CREATE VIEW SHIFTVIEW
AS
      SELECT
           Area,
           Shift_ID,
           Sequence,
           StartTime,
           dateadd(ms,-3,cast(isnull((SELECT top 1 StartTime FROM gen_shift WHERE Area = s.Area AND StartTime>s.StartTime),(SELECT top 1 dateadd(dd,1,StartTime) FROM Gen_Shift where Area= s.Area order by starttime))as datetime))   AS EndTime,
           datediff(hh,Starttime,isnull((SELECT top 1 StartTime FROM gen_shift WHERE Area = s.Area AND StartTime>s.StartTime),(SELECT top 1 dateadd(dd,1,StartTime) FROM Gen_Shift where Area= s.Area order by starttime))) as hours
      FROM gen_shift s
