create view [dbo].[ARC_STATBALANCE] as


SELECT IDType,Meta_ID,PtsBalanceForward,PtsPromoAdjusted,PtsMgrAdjusted,
PtsRedeemed,DollarsBalanceForward,DollarsPromoAdjusted,DollarsMgrAdjusted,
DollarsRedeemed,CompBalanceForward,CompPromoAdjusted,CompMgrAdjusted,
CompRedeemed,BonusBucksBalanceForward,BonusBucksPromoAdjusted,BonusBucksMgrAdjusted,
BonusBucksRedeemed,ExtraBucket1BalanceForward,ExtraBucket1PromoAdjusted,
ExtraBucket1MgrAdjusted,ExtraBucket1Redeemed,ExtraBucket2BalanceForward,
ExtraBucket2PromoAdjusted,ExtraBucket2MgrAdjusted,ExtraBucket2Redeemed,
ArchivedPts,ArchivedComp,RecoveredPts,RecoveredComp,ExpiredPts,ExpiredComp,a.PtsLost,a.CompsLost
from WinOasis_Arch.dbo.cds_statbalance b
left join WinOasis_Arch.dbo.ARC_LOST_POINTS_COMPS a on b.Meta_ID = a.Player_ID
