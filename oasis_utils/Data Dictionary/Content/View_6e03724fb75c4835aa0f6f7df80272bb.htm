CREATE VIEW [dbo].[CouponHistoryPivot2_vw]
AS
SELECT   cv.CouponID,
           cv.ValidationNumber AS ValidationNumber,
           cv.CouponSeriesID,
           cv.Amount AS Amount,
           cv.Status AS Status,
           ch.HistoryDate AS StatusDateTime,
           ci.HistoryDate AS IssuedDateTime,
           ci.UserID AS IssuedUserID,
           ci.AppID AS IssuedAppID,
           ci.SlotMastID AS IssuedSlotID,
         ci.Revision AS IssuedRevision,
         ci.LocationID AS IssuedLocationID,
         cs.TransactionDate AS PrintedDateTime,
         cs.UserID AS PrintedUserID,
         cs.AppID AS PrintedAppID,
         cs.SlotMastID AS PrintedSlotID,
         cs.Revision AS PrintedRevision,
         cs.LocationID AS PrintedLocationID,
           cs.CardNumber AS PrintedOCR,
           cr.TransactionDate AS RedeemedDateTime,
         cr.UserID AS RedeemedUserID,
         cr.AppID AS RedeemedAppID,
         cr.SlotMastID AS RedeemedSlotID,
         cr.Revision RedeemedRevision,
         cr.LocationID AS RedeemedLocationID,
         (CASE WHEN cr.Flags IN(25,26,27) THEN 1 ELSE 0 END) AS RedeemedOverride,
           cr.CardNumber AS RedeemedOCR,
         ce.TransactionDate AS ExpiredDateTime,
         ce.UserID AS ExpiredUserID,
         ce.AppID AS ExpiredAppID,
         ce.SlotMastID AS ExpiredSlotID,
         ce.Revision AS ExpiredRevision,
         ce.LocationID AS ExpiredLocationID,
         cvd.TransactionDate AS VoidedDateTime,  
         cvd.UserID AS VoidedUserID,
         cvd.AppID VoidedAppID,
         cvd.SlotMastID AS VoidedSlotID,
         cvd.Revision AS VoidedRevision,
         cvd.LocationID AS VoidedLocationID,
         cv.TicketNumber
FROM CouponHistoryRows chr
      INNER JOIN CouponView cv
           ON cv.CouponID = chr.CouponID
      INNER JOIN CouponHistory ch
           ON cv.CouponHistoryID = ch.CouponHistoryID
      LEFT JOIN CouponHistory cc      -- Created (0)
           ON cc.CouponHistoryID = chr.CouponCreatedID
      LEFT JOIN CouponHistory cs      -- Printed (2)
           ON cs.CouponHistoryID = chr.CouponSlotID
      LEFT JOIN CouponHistory cr      -- Redeemed (4)
           ON cr.CouponHistoryID = chr.CouponRedeemedID
      LEFT JOIN CouponHistory ce      -- Expired (5)
           ON ce.CouponHistoryID = chr.CouponExpiredID
      LEFT JOIN CouponHistory cvd      -- Voided (6)
           ON cvd.CouponHistoryID = chr.CouponVoidedID
      LEFT JOIN CouponHistory ci      -- Issued (8)
           on ci.CouponHistoryID = chr.CouponIssuedID
