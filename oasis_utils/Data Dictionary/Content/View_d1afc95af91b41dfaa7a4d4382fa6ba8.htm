DDL Code  CREATE VIEW dbo.StatDetailGroup_vw AS
SELECT s.Detail_ID, ge.Group_ID, s.Meta_ID Player_ID, s.OCR, s.StatType, s.SlotNumber, s.Location, s.Pit, s.TableNumber, s.Shift_ID, s.GamingDate, s.StartTime, s.EndTime, s.TimePlayed, s.PlayerSkill, s.PlayerSpeed, s.Denomination, s.RawAverageBetTimesTime1, s.RawAverageBetTimesTime2, s.RawAverageBetTimesTime3, s.MoneyPlay, 0 , s.FrontIn, s.ChipsIn, s.CreditIn, s.CashOut, s.JackPot, s.TWin, s.PtsEarned, s.PromoPtsEarned, s.DollarsEarned, s.PromoDollarsEarned, s.CompEarned, s.ExtraBucket1, s.ExtraBucket2, s.User_ID, s.RatedByUser_ID, s.PointCalc_ID
FROM dbo.CDS_STATDETAIL s (nolock)
     INNER JOIN dbo.CDS_GROUPENROLL ge (nolock) on ge.Player_ID=s.Meta_ID and ge.Active='Y' and ge.HasSubscribed='Y'
     INNER JOIN dbo.CDS_GROUP g (nolock) ON g.Group_ID = ge.Group_ID
WHERE s.IdType = 'P'
     AND s.StartTime >= ge.StartDatetime
     AND s.StartTime >= g.StartDatetime
     AND s.EndTime <= g.EndDatetime
