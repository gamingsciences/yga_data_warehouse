create view dbo.ATI_StatView
as
   select Meta_ID Player_ID, StatType,
       GamingDate, StartTime, EndTime, TimePlayed, RawAverageBetTimesTime1, RawAverageBetTimesTime2, RawAverageBetTimesTime3,
       MoneyPlay, Cashin, FrontIn, ChipsIn, CreditIn, CashOut, Jackpot, TWin, PtsEarned, CompEarned
     from CDS_STATDETAIL (nolock)
     where IDType = 'P'
   union all
   select PlayerID, 'SLOT',
       GamingDate, StartTime, EndTime, TimePlayed, 0 RawAverageBetTimesTime1, 0 RawAverageBetTimesTime2, 0 RawAverageBetTimesTime3,
       0 MoneyPlay, Cashin, 0 FrontIn, 0 ChipsIn, 0 CreditIn, CashOut, Jackpot, TWin, PtsEarned, CompEarned
     from ATI_StatSlot (nolock)
   union all
   select PlayerID, StatType,
       GamingDate, StartTime, EndTime, TimePlayed, RawAverageBetTimesTime1, RawAverageBetTimesTime2, RawAverageBetTimesTime3,
       MoneyPlay, Cashin, FrontIn, ChipsIn, CreditIn, CashOut, Jackpot, TWin, PtsEarned, CompEarned
     from ATI_StatPit (nolock)
