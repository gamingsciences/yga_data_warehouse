/**********************************************************************
NAME:                TicketHistoryPivot2_vw
Created By:           David W. Steyer
Date:                2/22/2006
Version:           11.6.0

View to mimick CDS_TICKETDETAIL and CDS_TICKET Structure for Reporting
purposes. Should be used in the Short time only.

***********************************************************************/
CREATE VIEW [dbo].[TicketHistoryPivot2_vw]
AS

SELECT      tv.TicketID,
           tv.ValidationNumber AS ValidationNumber,
           tv.TicketNumber as TicketNumber,
           tv.Amount AS Amount,
           tv.Status AS Status,
           th.HistoryDate AS StatusDateTime,
           ti.TransactionDate AS IssuedDateTime,
           ti.UserID AS IssuedUserID,
           ti.AppID AS IssuedAppID,
           ti.SlotMastID AS IssuedSlotID,
           ti.Revision AS IssuedRevision,
           ti.LocationID AS IssuedLocationID,
         ts.TransactionDate AS PrintedDateTime,
         ts.UserID AS PrintedUserID,
         ts.AppID AS PrintedAppID,
         ts.SlotMastID AS PrintedSlotID,
         ts.Revision AS PrintedRevision,
         ts.LocationID AS PrintedLocationID,
           ts.CardNumber AS PrintedOCR,
           tr.TransactionDate AS RedeemedDateTime,
         tr.UserID AS RedeemedUserID,
         tr.AppID AS RedeemedAppID,
         tr.SlotMastID AS RedeemedSlotID,
           tr.Revision AS RedeemedRevision,
         tr.LocationID AS RedeemedLocationID,
         (CASE WHEN tr.Flags IN(25,26,27) THEN 1 ELSE 0 END) AS RedeemedOverride,
           tr.OverrideID AS OverrideID,
           tr.CardNumber AS RedeemedOCR,
         te.TransactionDate AS ExpiredDateTime,
         te.UserID AS ExpiredUserID,
         te.AppID AS ExpiredAppID,
         te.SlotMastID AS ExpiredSlotID,
         te.Revision AS ExpiredRevision,
         te.LocationID AS ExpiredLocationID,
         tvd.TransactionDate AS VoidedDateTime,  
         tvd.UserID AS VoidedUserID,
         tvd.AppID AS VoidedAppID,
         tvd.SlotMastID AS VoidedSlotID,
         tvd.Revision AS VoidedRevision,
         tvd.LocationID AS VoidedLocationID

FROM TicketHistoryRows thr
      INNER JOIN TicketView tv
           ON tv.TicketID = thr.TicketID
      INNER JOIN TicketHistory th
           ON tv.TicketHistoryID = th.TicketHistoryID
      LEFT JOIN TicketHistory tc      -- Created (0)
           ON tc.TicketHistoryID = thr.TicketCreatedID
      LEFT JOIN TicketHistory ts      -- Printed (2)
           ON ts.TicketHistoryID = thr.TicketSlotID
      LEFT JOIN TicketHistory tr      -- Redeemed (4)
           ON tr.TicketHistoryID = thr.TicketRedeemedID
      LEFT JOIN TicketHistory te      -- Expired (5)
           ON te.TicketHistoryID = thr.TicketExpiredID
      LEFT JOIN TicketHistory tvd      -- Voided (6)
           ON tvd.TicketHistoryID=thr.TicketVoidedID
      LEFT JOIN TicketHistory ti      -- Issued (8)
           on ti.TicketHistoryID = thr.TicketIssuedID
