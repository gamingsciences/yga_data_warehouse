   SELECT

     p.CustomFlag1,p.CustomFlag2,p.CustomFlag3,p.CustomFlag4,p.CustomFlag5,p.CreditLimit,p.TTOLimit,l.AllowComboRedeem,p.MailFlag,  
     a.ResidenceSameAsMailing,l.AllowComboCredit
   FROM WinOasis_Arch.dbo.CDS_PLAYER p
     LEFT OUTER JOIN   WinOasis_Arch.dbo.PlayerAddress a ON p.Player_ID = a.Player_ID
