DDL Code  CREATE VIEW dbo.CDS_BUFFERVIEW AS
SELECT
   b.TransCode,b.TransDate,
   b.SessionPt,b.Resid,b.BonusPt,
   ct.Tx_ID as cttx,jpt.tx_ID as jptx,tk.Tx_ID as tktx,tx.Tx_ID as txtx,
   COALESCE(ct.Tx_ID,jpt.tx_ID,tk.Tx_ID,tx.Tx_ID,0) AS Tx_ID,
   COALESCE(ct.Description,jpt.Description,tk.Description,tx.Description,'Unknown') AS Description,
   b.RB1,b.RB2,b.RB3,B.RB4,b.RW1,b.RW2,
   b.SB1,b.SB2,b.SB3,b.SB4,b.SW1,b.SW2,

   b.RecordNum, b.PollerID, b.[TimeStamp],
   b.CoinIn, b.CoinOut, b.OCR, b.SlotMast_ID, b.Revision
--   c.Site, c.Dpu, c.Sen, c.MacAddress,   c.SentinelTypeID,
--   c.SlotNumber, -- c.Multiplier,
--   c.Denomination, c.Par, CDS_BUFFER.NCEPPlayed
FROM
(
     SELECT SessionPt,Resid,BonusPt,
     CASE   TransCode
     WHEN 93 THEN
     CAST(SUBSTRING(CAST(ResID as Binary(4)),3,1) as tinyint)               
     WHEN 94 THEN
       CAST(Sessionpt %65535 as Smallint)               
     WHEN 107 THEN
       CAST(SUBSTRING(CAST(Sessionpt as Binary(4)),2,1) as smallint)
     ELSE
       CAST(SUBSTRING(CAST(ResID as Binary(4)),4,1) as smallint)
     END AS Tx_ID,

     CAST(SUBSTRING(CAST(SessionPt as Binary(4)),1,1) as tinyint) AS SB4,
     CAST(SUBSTRING(CAST(SessionPt as Binary(4)),2,1) as tinyint) AS SB3,
     CAST(SUBSTRING(CAST(SessionPt as Binary(4)),3,1) as tinyint) AS SB2,
     CAST(SUBSTRING(CAST(SessionPt as Binary(4)),4,1) as tinyint) AS SB1,
     CAST(SUBSTRING(CAST(SessionPt as Binary(4)),3,2) as smallint) AS SW1,
     CAST(SUBSTRING(CAST(SessionPt as Binary(4)),1,2) as smallint) AS SW2,
     CAST(SUBSTRING(CAST(ResID as Binary(4)),1,1) as tinyint) AS RB4,
     CAST(SUBSTRING(CAST(ResID as Binary(4)),2,1) as tinyint) AS RB3,
     CAST(SUBSTRING(CAST(ResID as Binary(4)),3,1) as tinyint) AS RB2,
     CAST(SUBSTRING(CAST(ResID as Binary(4)),4,1) as tinyint) AS RB1,
     CAST(SUBSTRING(CAST(ResID as Binary(4)),3,2) as smallint) AS RW1,
     CAST(SUBSTRING(CAST(ResID as Binary(4)),1,2) as smallint) AS RW2,

     RecordNum,PollerID,
     TransDate,TransCode,
     CoinIn,CoinOut,OCR,SlotMast_ID,Revision,
     [TimeStamp]
     
      FROM dbo.CDS_BUFFER (NOLOCK)
) b
--   INNER JOIN dbo.CDS_SlotMast c (NOLOCK) ON c.SlotMast_ID = b.SlotMast_ID AND c.Revision = b.Revision
   LEFT JOIN
   (
           SELECT 128 AS Tx_ID,'Machine Jackpot' AS Description
           UNION SELECT 128 AS Tx_ID,'Machine Jackpot'
           UNION SELECT 129,'Cancelled Credit'
           UNION SELECT 130,'Top Award'
           UNION SELECT 0,'Progressive Level 0'
           UNION SELECT 1,'Progressive Level 1'
           UNION SELECT 2,'Progressive Level 2'
           UNION SELECT 3,'Progressive Level 3'
           UNION SELECT 4,'Progressive Level 4'
           UNION SELECT 5,'Progressive Level 5'
           UNION SELECT 6,'Progressive Level 6'
           UNION SELECT 7,'Progressive Level 7'
           UNION SELECT 8,'Progressive Level 8'
           UNION SELECT 9,'Progressive Level 9'
      )jpt ON jpt.Tx_ID = b.Tx_ID AND b.TransCode = 93
      LEFT JOIN
      (
           SELECT 2 as Tx_ID,'Ticket Inserted, Vend Requested' as Description
           UNION SELECT   3,'Request Ticket Information'
           UNION SELECT   16,'Ticket Processing'
           UNION SELECT   17,'Ticket Printed, System Val #'
           UNION SELECT   18,'Ticket Printed'
           UNION SELECT   19,'Ticket Accepted, Vend Complete'
           UNION SELECT   21,'Comp Ticket Printed'
           UNION SELECT   23,'Machine rejected voucher'
           UNION SELECT   24,'Ticket confiscated by machine'
           UNION SELECT   32,'Print status unknown'
           UNION SELECT   33,'Game timeout awaiting val #'
           UNION SELECT   34,'Invalid print message'
           UNION SELECT   47,'Printer failure, unable to print'
           UNION SELECT   48,'Unknow vend ticket status'
           UNION SELECT   49,'Reply timeout'
           UNION SELECT   50,'Incompatible denomination'
           UNION SELECT   51,'Ticket over limit'
           UNION SELECT   52,'Ticket over credit limit'
           UNION SELECT   53,'Ticket rejected by game'
           UNION SELECT   63,'Validator failure'
           UNION SELECT   192,'Request system validation #'
           UNION SELECT   233,'Vend request denied by DPU'
           UNION SELECT   234,'Val # request denied by DPU'
      ) tk ON tk.Tx_ID = b.Tx_ID AND b.TransCode=107
      LEFT JOIN dbo.CDS_COMPOSITE_TILTS_EXT ct (nolock) ON   ct.tx_ID = b.Tx_ID AND b.TransCode = 94
      LEFT JOIN dbo.CDS_TXCODES tx (nolock) ON tx.Tx_ID = b.TransCode AND b.TransCode Not IN(93,94,107)
